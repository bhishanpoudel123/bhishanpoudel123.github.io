from sklearn.compose import ColumnTransformer
from sklearn.pipeline import Pipeline

cols_oe = ['sex','adult_male','alone']
cols_ohe = ['pclass','embarked']

oe = OrdinalEncoder()
ohe = OneHotEncoder(handle_unknown='ignore')

imp = SimpleImputer(add_indicator=False) # numerical
imp_freq = SimpleImputer(strategy='most_frequent') # ordinal and ohe

pipe_imp_oe = Pipeline([('imp_freq', imp_freq), ('oe_bin', oe)])
pipe_imp_ohe = Pipeline([('imp_freq', imp_freq), ('ohe_cat', ohe)])

ct = ColumnTransformer(
    [
     # clip and impute fare
     ('0_clip_fare_imp',
      Pipeline([('clip_fare', clip_fare), ('imp', imp)]),
      ['fare']),
     
     # get first letter, impute, one hot encode deck
     ('1_firstLetter_impFreq_ohe_deck',
      Pipeline([('get_first_letter', get_first_letter),
                ('imp_freq', imp_freq),
                ('ohe_deck', ohe)
                ]),
      ['deck']),

    # ordinal encode binary labels
    ('2_pipe_imp_oe', pipe_imp_oe, cols_oe),

    # one hot encode category cols
    ('3_pipe_imp_ohe', pipe_imp_ohe, cols_ohe),
     
    # impute age
    ('4_imp_age', imp, ['age']),

    ],
    remainder='drop')

pipe = Pipeline([('preprocessor', ct), ('clf', clf)])
pipe

def plot_roc_curve(fpr, tpr,my_auc):
    plt.plot(fpr, tpr, color='orange', label=f'ROC: {my_auc:.4f}')
    plt.plot([0, 1], [0, 1], color='darkblue', linestyle='--')
    plt.xlabel('False Positive Rate')
    plt.ylabel('True Positive Rate')
    plt.title('Receiver Operating Characteristic (ROC) Curve')
    plt.legend()
    plt.show()

optimal_idx = np.argmax(tpr - fpr)
optimal_threshold = thresholds[optimal_idx]

from sklearn.experimental import enable_hist_gradient_boosting
from sklearn.ensemble import HistGradientBoostingClassifier
